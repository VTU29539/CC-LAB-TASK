#include <stdio.h> 
#include <stdlib.h> 
#define MAX_N 10 
int n; 
int board[MAX_N][MAX_N]; 
int is_valid(int row, int col) { 
    int i, j; 
    for (i = 0; i < n; i++) { 
        if (board[row][i] || board[i][col]) return 0; 
    } 
    for (i = row, j = col; i >= 0 && j >= 0; i--, j--) { 
        if (board[i][j]) return 0; 
    } 
    for (i = row, j = col; i < n && j >= 0; i++, j--) { 
        if (board[i][j]) return 0; 
    } 
    return 1; 
} 
int solve(int col) { 
    if (col >= n) return 1; 
    for (int row = 0; row < n; row++) { 
        if (is_valid(row, col)) { 
            board[row][col] = 1; 
            if (solve(col + 1)) return 1; 
            board[row][col] = 0; 
        } 
    } 
    return 0; 
} 
int main() { 
    scanf("%d", &n); 
    if (solve(0)) { 
        for (int i = 0; i < n; i++) { 
            for (int j = 0; j < n; j++) { 
                printf("%d ", board[i][j]); 
            } 
            printf("\n"); 
        } 
    } else { 
        printf("Not possible\n"); 
    } 
    return 0; 
}
